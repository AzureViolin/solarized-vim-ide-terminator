# solarized-vim-ide-terminator

```bash
cp -r terminator ~/.config/
cp .vimrc ~/
mkdir ~/.vim
cp -r bundle ~/.vim/
cp -r fonts ~/.local/share
sudo cp -r conf.avail /usr/share/fontconfig/

cd /etc/fonts/conf.d
sudo ls -sf ../conf.aval/10-powerline-symbols.conf 10-powerline-symbols.conf

fc-cache -fv
```
